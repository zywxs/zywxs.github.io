  • 创建刷新配置脚本
    由于配置过程中会多次更新配置文件干脆写一个重置脚本
    #!/bin/sh
    systemctl stop slapd
    rm -rf /etc/openldap/slapd.d/*
    slaptest -f /etc/openldap/slapd.conf -F /etc/openldap/slapd.d/
    chown -R ldap.ldap /etc/openldap/slapd.d/
    systemctl restart slapd
    
    bash reset_ldap.sh
